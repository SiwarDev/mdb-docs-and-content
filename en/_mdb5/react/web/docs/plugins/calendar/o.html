MDB calendar is a plugin that allows you to efficiently manage tasks. Thanks to this extension you will be able to easily create new events, manage current events, move existing events between other days, and store all data in an easily readable array.

Note: Read the API tab to find all available options and advanced customization

Basic example

A few predefined events allowing you to see how the plugin looks like.

import React, { useState } from 'react'; import { MDBCalendar } from 'mdb-react-calendar'; export default function App(): JSX.Element { const [myEvents, setMyEvents] = useState([ { summary: 'Angular Meetup', description: 'Duis aute irure dolor in reprehenderit in voluptate velit esse cillum dolore eu fugiat nulla pariatur', start: { date: '19/09/2021', time: '10:00', }, end: { date: '25/09/2021', time: '14:00', }, color: { background: '#F93154', foreground: 'white', }, }, { summary: 'JS Conference', description: '', start: { date: '22/09/2021', }, end: { date: '22/09/2021', }, color: { background: '#1266F1', foreground: 'white', }, }, { summary: 'Vue Meetup', description: '', start: { date: '23/09/2021', }, end: { date: '28/09/2021', }, color: { background: '#00B74A', foreground: 'white', }, }, { summary: 'React Meetup', description: '', start: { date: '29/09/2021', }, end: { date: '30/09/2021', }, color: { background: '#39C0ED', foreground: 'white', }, }, { summary: 'Meeting', description: '', start: { date: '13/09/2021', time: '8:00', }, end: { date: '13/09/2021', time: '12:00', }, color: { background: '#1266F1', foreground: 'white', }, }, { summary: 'Call', description: '', start: { date: '14/09/2021', time: '11:00', }, end: { date: '14/09/2021', time: '14:00', }, color: { background: '#FFA900', foreground: 'white', }, }, ]); return ( ); }
